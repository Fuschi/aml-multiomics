prepare_data

Reads raw files from all three cohorts, aligns samples to WES IDs, and writes clean matrices to processed/.

library(tidyverse)
library(knitr)

WES mutations

Long-format somatic mutation table shared across BEAT-AML, TCGA, and NGS-PTL. TCGA sample IDs have their -03 suffix stripped to match the format used in the expression data.

wes <- read_xlsx("../raw/WES_filtered_final.xlsx") |>
  mutate(ID = if_else(cohort == "TCGA", str_remove(ID, "-03"), ID))

BEAT-AML expression

The expression matrix uses RNA-seq IDs (dbgap_rnaseq_sample). The sample mapping table (source: Bottomly et al. 2022) converts these to WES IDs (labId), keeping only samples flagged for inclusion.

expr_beat <- read_tsv("../raw/210126_BEAT-AML_CPM-TMM_log2_with_hugo.tsv.xz",
                      show_col_types = FALSE) |>
  column_to_rownames("symbol") |>
  t() |>
  as_tibble(rownames = "dbgap_rnaseq_sample")

meta_beat <- read_xlsx("../raw/beataml_waves1to4_sample_mapping.xlsx") |>
  filter(rna_include_in_analysis == "yes")

expr_beat <- expr_beat |>
  left_join(select(meta_beat, dbgap_rnaseq_sample, labId),
            by = "dbgap_rnaseq_sample") |>
  relocate(labId, .after = "dbgap_rnaseq_sample") |>
  select(-dbgap_rnaseq_sample)

Keep only samples present in both expression and WES.

sample_id_wes <- wes |> filter(cohort == "BEAT-AML") |> pull(ID) |> unique()
ggVennDiagram(x = list(wes = sample_id_wes, rna = expr_beat$labId))
rm(sample_id_wes)
expr_beat_aml <- expr_beat |>
  filter(labId %in% intersect(unique(wes$ID), labId))

Metadata

Sample-level metadata assembled from the karyotype file and the BEAT-AML mapping table. Rows are sorted to match the expression matrix row order.

karyotype <- read_xlsx("../raw/Karyotype and other variables.xlsx", col_types = "text") |>
  select(-2, -6) |>
  rename(Disease_Type = Disease.type, sample_id = ID)
beat_aml_metadata <- meta_beat |>
  filter(labId %in% expr_beat_aml$labId) |>
  relocate(labId) |>
  rename(sample_id = labId) |>
  left_join(
    karyotype |>
      filter(cohort == "BEAT-AML") |>
      select(sample_id, Patient_Id, Gender, Disease_Type),
    by = "sample_id"
  ) |>
  arrange(match(sample_id, expr_beat_aml$labId))

tcga_aml_metadata <- karyotype |>
  filter(cohort == "TCGA") |>
  filter(sample_id %in% expr_tcga_aml$sample_id) |>
  arrange(match(sample_id, expr_tcga_aml$sample_id))

write_tsv(beat_aml_metadata,  "../processed/beat_aml_metadata.tsv")
write_tsv(tcga_aml_metadata, "../processed/tcga_aml_metadata.tsv")

Human-GEM: metabolic gene annotation

Human-GEM links genes to metabolic subsystems through GPR (gene–protein–reaction) rules encoded as Boolean expressions with Ensembl IDs. We extract a flat gene → subsystem table used downstream to filter expression matrices to metabolic genes before WGCNA.

gem_genes <- read_xlsx("../raw/Human-GEM.xlsx", sheet = "GENES") |>
  select(ensembl_id = NAME, hugo = `SHORT NAME`)

gem_rxns <- read_xlsx("../raw/Human-GEM.xlsx", sheet = "RXNS") |>
  select(reaction_id = ID,
         reaction_name = NAME,
         gene_association = `GENE ASSOCIATION`,
         subsystem = SUBSYSTEM) |>
  filter(!is.na(gene_association))

gem <- gem_rxns |>
  mutate(ensembl_id = str_extract_all(gene_association, "ENSG\\d+")) |>
  unnest(ensembl_id) |>
  left_join(gem_genes, by = "ensembl_id") |>
  filter(!is.na(hugo)) |>
  select(hugo, ensembl_id, reaction_id, reaction_name, subsystem) |>
  distinct()

gem_by_gene <- gem |>
  group_by(hugo, ensembl_id) |>
  summarise(
    n_reactions  = n(),
    n_subsystems = n_distinct(subsystem),
    subsystems   = paste(unique(subsystem), collapse = ";"),
    reactions    = paste(unique(reaction_id), collapse = ";"),
    .groups = "drop"
  )

write_tsv(gem_by_gene, "../processed/gem_gene_subsystems.tsv")
