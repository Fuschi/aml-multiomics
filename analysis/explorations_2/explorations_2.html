We load pre-processed expression matrices for BEAT-AML (217 samples, 22,000+ genes, log2 TMM-CPM normalized, Hugo symbols) and TCGA-LAML (140 samples, 30,000+ genes, linear TMM-CPM normalized, Hugo symbols), restricted to samples with matched WES information. Gene-level metabolic annotation from Human-GEM restricts downstream analyses to ~2,200 metabolic genes linked to known reactions and subsystems.

top_pathway_class <- long_aml |>
  select(sample_id, pathway_class, log2_tmm_cpm) |>
  filter(pathway_class != "non_metabolic") |>
  group_by(sample_id, pathway_class) |>
  summarise(abun_metab = sum(log2_tmm_cpm), .groups = "drop") |>
  group_by(sample_id) |>
  mutate(rela_metab = abun_metab / sum(abun_metab)) |>
  ungroup() |>
  group_by(pathway_class) |>
  summarise(total_rela = sum(rela_metab), .groups = "drop") |>
  arrange(desc(total_rela)) %>%
  slice_max(total_rela, n = 20) %>%
  pull(pathway_class) 
pathway_order <- c(top_pathway_class, "other_metabolic", "non_metabolic")

long_aml <- long_aml %>%
  mutate(pathway_class_fct = case_when(

  mutate(pathway_class_fct = factor(pathway_class_fct, levels = pathway_order)) %>%

plot_stacked_bar <- function(data,
                             include_non_metabolic = TRUE,
                             relative              = FALSE,
                             title                 = NULL) {

  if (!include_non_metabolic) {
    data <- data |> filter(pathway_class_fct != "non_metabolic")
  }

  data <- data |>
    group_by(cohort, sample_id, pathway_class_fct) |>
    summarise(log2_tmm_cpm = sum(log2_tmm_cpm), .groups = "drop")

  if (relative) {
    data <- data |>
      group_by(cohort, sample_id) |>
      mutate(y_var = log2_tmm_cpm / sum(log2_tmm_cpm)) |>
      ungroup()
    y_label <- "Relative abundance"
  } else {
    data <- data |> mutate(y_var = log2_tmm_cpm)
    y_label <- "log2(TMM-CPM+1)"
  }

  data |>
    ggplot(aes(x = sample_id, y = y_var, fill = pathway_class_fct)) +
    geom_col() +
    theme_minimal() +
    theme(axis.text.x     = element_blank(),
          axis.ticks.x    = element_blank(),
          legend.position = "top") +
    labs(x     = "Samples",
         y     = y_label,
         fill  = "Pathway class",
         title = title) +
    facet_wrap(~cohort, nrow = 2, scales = "free") +
    ggqualpal::scale_fill_qualpal()
}
